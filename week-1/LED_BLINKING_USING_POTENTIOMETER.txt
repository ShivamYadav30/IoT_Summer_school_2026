int LED = 8;
int LDR = A5;
int ldr_value = 0;


void setup(){
   pinMode(LED, OUTPUT);
 Serial.begin(9600);

}
void loop(){
 ldr_value = analogRead(LDR);
   Serial.print("Light Intensity : ");
   Serial.println(ldr_value);
   digitalWrite(LED, HIGH);
  delay(500);
   digitalWrite(LED, LOW);
  delay(500);


}
