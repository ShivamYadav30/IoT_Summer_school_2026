#include <Ultrasonic.h>
int trig = 6;
int echo = 7;
int red = 3;
int yellow = 4;
int green = 5;
int buzzer = 8;
long distance;
float time;

void setup() {
  
  pinMode(red, OUTPUT);
  pinMode(yellow, OUTPUT);
  pinMode(yellow, OUTPUT);
  pinMode(green, OUTPUT);
  pinMode(trig, INPUT);
  pinMode(echo, INPUT);
  pinMode(buzzer, OUTPUT);


  Serial.begin(9600);
  

}
void loop(){
   distance = (time*0.034)/2;

   digitalWrite(trig, LOW);
   delay(2);

   digitalWrite(trig, HIGH);
   delay(10);
   digitalWrite(trig,LOW);

   time = pulseIn(echo,HIGH);

   distance = (time*0.034)/2;

   Serial.print("distance from the wall :");
   Serial.print(distance);
   Serial.println(" cm.");

   if(distance > 50 ){
    digitalWrite(green, HIGH);
    digitalWrite(yellow, LOW);
    digitalWrite(red, LOW);
    Serial.println("SAFE");
    
    delay(2000);


    digitalWrite(buzzer, HIGH);
    delay(1000);
    digitalWrite(buzzer, LOW);
    delay(1000);

   }
   else if(distance > 20 ){
    Serial.println("CAUTION");
    digitalWrite(yellow, HIGH);
    digitalWrite(green, LOW);
    digitalWrite(red, LOW);
    delay(500);
  
  
  digitalWrite(buzzer, HIGH);
  delay(500);
  digitalWrite(buzzer, LOW);
  delay(500);

   }
else if( distance > 10){
    digitalWrite(red, HIGH);
    digitalWrite(green, LOW);
    digitalWrite(yellow,LOW);
    Serial.println("WARNING, NOT SAFE");
    delay(200);
    digitalWrite(buzzer, HIGH);
    delay(200);
    digitalWrite(buzzer, LOW);
    delay(200);

    }
else {
     Serial.println( "STOP! IT'S A DANGER ZONE");
     digitalWrite(buzzer, HIGH);
     digitalWrite(red, HIGH);
     digitalWrite(yellow, HIGH);
     digitalWrite(green, HIGH);
     delay(100);
     digitalWrite(buzzer, LOW);
     digitalWrite(red, HIGH);
     digitalWrite(yellow, LOW);
     digitalWrite(green, LOW);
     
    }

  
}
