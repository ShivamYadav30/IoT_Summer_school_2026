#include <Servo.h>

const int POT_PIN = A0;      
const int SERVO_PIN = 9;    
const int BUTTON_PIN = 2;    
Servo myServo;

int lastButtonState = HIGH;  
unsigned long lastDebounceTime = 0;
const unsigned long DEBOUNCE_DELAY = 50; 

void setup() {
  Serial.begin(9600);
  
  myServo.attach(SERVO_PIN);
  
   pinMode(BUTTON_PIN, INPUT_PULLUP); 
  
   int initialPot = analogRead(POT_PIN);
  int initialAngle = map(initialPot, 0, 1023, 0, 180);
  myServo.write(initialAngle);
}

void loop() {
  int reading = digitalRead(BUTTON_PIN);
  
   if (reading != lastButtonState) {
    lastDebounceTime = millis();
  }
  
  if ((millis() - lastDebounceTime) > DEBOUNCE_DELAY) {
       if (reading == LOW && lastButtonState == HIGH) {
      executeSweep();
    }
  }
  
  lastButtonState = reading;

  int potValue = analogRead(POT_PIN);
 )
  int targetAngle = map(potValue, 0, 1023, 0, 180);
  
  myServo.write(targetAngle);
  

  Serial.print("Servo Angle: ");
  Serial.print(targetAngle);     
  Serial.println(" degrees");
  
  delay(15); 
}


void executeSweep() {
  Serial.println("--- Starting Sweep Sequence ---");
  
  
  for (int angle = 0; angle <= 180; angle++) {
    myServo.write(angle);
    Serial.print("Sweeping up: ");
    Serial.print(angle);          // Fixed
    Serial.println(" degrees");
    delay(15); 
  }
  
  delay(200); 
  
  
  for (int angle = 180; angle >= 0; angle--) {
    myServo.write(angle);
    Serial.print("Sweeping down: ");
    Serial.print(angle);        // Fixed
    Serial.println(" degrees");
    delay(15); 
  }
  
  Serial.println("--- Sweep Complete! Returning to Potentiometer Control ---");
}
